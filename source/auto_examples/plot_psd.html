import matplotlib.pyplot as plt
import pysp2


my_sp2b = pysp2.io.read_sp2(pysp2.testing.EXAMPLE_SP2B)

my_hk = pysp2.io.read_hk_file(pysp2.testing.EXAMPLE_HK)
